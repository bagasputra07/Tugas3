import 'dart:convert';

class Mobil {
    String merk;
    int tahun;

    // Parameterized constructor
    Mobil(this.merk, this.tahun);

    // Named constructor .baru (tanpa parameter)
    Mobil.baru() {
        merk = 'Bmw';
        tahun = 2023;
    }

    // Named constructor .dariJson (untuk Map)
    Mobil.dariJson(Map data) {
        merk = data['merk'] ?? 'Toyota';
        tahun = data['tahun'] ?? 2020;
    }

    @override
    String toString() => 'Mobil{merk: $merk, tahun: $tahun}';
}

void main() {
    print('--- PRAKTIKUM KELAS MOBIL ---');

    var m1 = Mobil('Ferrari', 2022);
    print('Mobil 1 (default): $m1');

    var m2 = Mobil.baru();
    print('Mobil 2 (.baru): $m2');

    String dataJson = '{ "merk": "Lamborghini", "tahun": 2023 }';
    Map jsonData = jsonDecode(dataJson);
    var m3 = Mobil.dariJson(jsonData);
    print('Mobil 3 (.dariJson): $m3');

    Map dataLain = { 'merk': 'Porsche', 'tahun': 2024 };
    var m4 = Mobil.dariJson(dataLain);
    print('Mobil 4 (.dariJson): $m4');

    print('--- SELESAI DEMONSTRASI ---');
}
✅ Demonstrasi named constructor .baru dan .dariJson

    print(scalletLagi);

}

class Titik {
    final double x, y;

    // Menggunakan initializer list untuk validasi / transformasi
    Titik(double x, double y) : 
        x = x < 0 ? 0 : x,   // jika negatif jadikan 0
        y = y < 0 ? 0 : y {
            print('Titik dibuat di ($this.x, $this.y)');
        }
}

void main() {
    print('Membuat objek Titik pertama:');
    var t1 = Titik(8.0, 3.5);
    print('Titik 1: x = ${t1.x}, y = ${t1.y}');

    print('\nMembuat objek Titik kedua dengan nilai negatif:');
    var t2 = Titik(-8.0, 3.5);
    print('Titik 2: x = ${t2.x}, y = ${t2.y}');

    print('\nMembuat objek Titik ketiga dengan negatif di kedua koordinat:');
    var t3 = Titik(-2.0, -4.0);
    print('Titik 3: x = ${t3.x}, y = ${t3.y}');
}

class Produk {

}

class User {
    String _username;
    String _password;
    String _email;

    User(this._username, this._password, this._email);

    // Getter
    String get username => _username;
    String get password => _password;
    String get email => _email;

    // Setter dengan validasi
    set username(String v) {
        if (v.length >= 5) {
            _username = v;
        } else {
            print('Username minimal 5 karakter');
        }
    }

    set password(String v) {
        if (v.length >= 8) {
            _password = v;
        } else {
            print('Password minimal 8 karakter');
        }
    }

    set email(String v) {
        if (v.contains('@')) {
            _email = v;
        } else {
            print('Format email salah (Harus mengandung @)');
        }
    }

    void info() => print('User: $_username, Pass: $_password, Email: $_email');
}

void main() {
    var u1 = User('Bagas', '12345678', 'bagas.com');
    u1.username = 'BagasP';
    u1.password = 'thegreat1';
    u1.email = 'bagas@gmail.com';
    u1.info();
}

// Nama : Muhammad Bagas putra andeza

class User {
    String _username;
    String _password;

    User(this._username, this._password);

    String get username => _username;

    bool login(String username, String password) {
        return _username == username && _password == password;
    }

    void gantiPassword(String lama, String baru) {
        if (_password == lama) {
            if (baru.length >= 8) {
                _password = baru;
                print('Password berhasil diubah.');
            } else {
                print('Gagal: Password baru minimal 8 karakter.');
            }
        } else {
            print('Gagal: Password lama tidak cocok.');
        }
    }
}

void main() {
    print('--- MEMBUAT AKUN BARU ---');
    final User u1 = User('super_Bagas', 'rahasia123');

    print('--- PROSES LOGIN ---');
    if (u1.login('super_Bagas', 'rahasia123')) {
        print('✅ Selamat datang! Anda berhasil masuk.');
    }

    print('\n--- PERUBAHAN KATA SANDI ---');
    u1.gantiPassword('rahasia123', 'superSecret789'); // sukses
    u1.gantiPassword('salahPass', 'cobaCoba');        // gagal
}

class RekeningBank {
    String _nomorRekening;
    double _saldo;

    RekeningBank(this._nomorRekening, this._saldo);

    void deposit(double jml) {
        if (jml > 0) {
            _saldo += jml;
            print('Deposit Rp $jml Berhasil.');
        }
    }

    void tarik(double jml) {
        if (jml > 0 && _saldo >= jml) {
            _saldo -= jml;
            print('Tarik Rp $jml Berhasil.');
        } else {
            print('Gagal: Saldo tidak cukup atau input tidak valid.');
        }
    }

    void cekSaldo() {
        print('Rekening: $_nomorRekening | Saldo: Rp $_saldo');
    }
}

void main() {
    print('--- TRANSAKSI PERTAMA ---');
    var rekeningSaya = RekeningBank('1234-5678-90', 5000000);
    rekeningSaya.cekSaldo();

    print('\n--- MELAKUKAN PENARIKAN ---');
    rekeningSaya.tarik(1000000);
    rekeningSaya.cekSaldo();
}
