Multiple Constructor

class Mahasiswa{ String nama; String nim; Mahasiswa(this.nama,this.nim); Mahasiswa.tanpaNama(){ nama="Tanpa Nama"; nim="00000"; } void info(){ print("Mahasiswa: $nama - $nim"); } } void main(){ var m1 = Mahasiswa("Bagas","251410003"); var m2 = Mahasiswa.tanpaNama(); m1.info(); m2.info(); }

Getter dan Setter

class Produk{ String _nama; double _harga; Produk(this._nama,this._harga); String get nama => _nama; double get harga => _harga; set nama(String value){ if(value.length >=3){ _nama=value; } } set harga(double value){ if(value>0){ _harga=value; } } } void main(){ var p = Produk("Laptop",5000); print(p.nama); p.nama="Monitor"; print(p.nama); }

Latihan Bank

class RekeningBank{ String nomor; double saldo; RekeningBank(this.nomor,this.saldo); void deposit(double jumlah){ saldo += jumlah; } void tarik(double jumlah){ if(saldo>=jumlah){ saldo -= jumlah; } } void cekSaldo(){ print("Saldo: $saldo"); } } void main(){ var r = RekeningBank("123456",500000); r.deposit(200000); r.tarik(100000); r.cekSaldo(); }
